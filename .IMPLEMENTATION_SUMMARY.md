## 🎯 Key Implementation Details

### Save/Unsave Dream
```typescript
const handleSaveDream = async () => {
  if (isSaved) {
    removeDream(item.id)
    triggerToast('Removed from your Dream Board')
  } else {
    const dreamItem = { ...item, savedAt: new Date().toISOString() }
    addDream(dreamItem)
    triggerToast('Saved to your Dream Board ✨')
  }
}
```

### Share Functionality
```typescript
const handleShare = async () => {
  if (navigator.share) {
    await navigator.share(shareData) // Native share
  } else {
    await navigator.clipboard.writeText(url) // Fallback
    triggerToast('Link copied to clipboard!')
  }
}
```

### Toast Auto-Dismiss
```typescript
const triggerToast = useCallback((msg: string) => {
  if (toastTimer.current) clearTimeout(toastTimer.current)
  setToastMessage(msg)
  setShowToast(true)
  toastTimer.current = setTimeout(() => setShowToast(false), 3000)
}, [])
```

### Carousel Smooth Scroll
```typescript
const scrollCarousel = (dir: 'left' | 'right') => {
  carouselRef.current?.scrollBy({
    left: dir === 'left' ? -300 : 300,
    behavior: 'smooth'
  })
}
```
